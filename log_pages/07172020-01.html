    ---- system install ----
    vim /etc/pacman.d/mirrorlist    # dgut.edu.cn
    pacstrap /mnt base base-devel linux linux-firmware vim networkmanager dhcpcd


    ---- boot loader install ----
    pacman -S grub efibootmgr
    grub-install --efi-directory=/efi  --bootloader-id=Linux
    grub-mkconfig -o /boot/grub/grub.cfg


    ---- desktop environment install ----
    pacman -S xorg plasma plasma-applications  xf86-video-intel xf86-video-noureau


    ---- software ----
